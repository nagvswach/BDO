5.
SELECT DISTINCT s.SNAME
FROM s
JOIN spj ON spj.ID_S =s.ID_S
WHERE YEAR(spj.DATE_DERNIERE_LIVRAISON) BETWEEN 2015 AND 2019;

6.
SELECT s.ID_S,p.ID_P,j.ID_J 
from s,p,j
WHERE s.CITY <> p.CITY OR p.CITY <> j.CITY;

7.

FROM spj s1, spj s2
WHERE s1.ID_S = s2.ID_S AND s1.ID_P > s2.ID_P;
