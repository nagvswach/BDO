FROM spj
JOIN j ON SPJ.ID_J = j.ID_J
GROUP by j.ID_J,j.JNAME

4.
SELECT spj.ID_S,spj.ID_P,spj.ID_J,QTY * p.WEIGHT
FROM spj
JOIN p ON p.ID_P=spj.ID_P
GROUP by spj.ID_S,spj.ID_P,spj.ID_J;

FROM spj
JOIN p ON p.ID_P=spj.ID_P
